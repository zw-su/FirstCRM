class MyQueue:
    def __init__(self, maxsize=0):
        self.size = maxsize
        self.queue = []
        if type(maxsize) is not int:
            raise TypeError('传入参数类型错误')

    def qsize(self):
        return len(self.queue)

    def put(self, item):
        try:
            if not self.size:
                self.queue.append(item)
            else:
                if len(self.queue) >= self.size:
                    print('队列已满,数据:%s,未存放' % item)
                else:
                    self.queue.append(item)
        except TypeError as e:
            print(e)

    def get(self):
        if len(self.queue) > 0:
            return self.queue.pop(0)

# 编程题
from multiprocessing import Pool, Queue

Q = Queue()


def factorial(n):
    amount = 1
    for i in range(1, n + 1):
        amount *= i
    Q.put(amount)


def counter(n):
    sum = 0
    for i in range(n):
        aa = Q.get()
        sum += aa
    print('>>sum', sum)


def main(n, process_num=1):
    pool = Pool(processes=process_num)
    pool.apply_async(counter, (n,))
    for i in range(1, n + 1):
        pool.apply_async(factorial, (i,))
    pool.close()
    pool.join()


if __name__ == '__main__':
    main(10000, 8)
